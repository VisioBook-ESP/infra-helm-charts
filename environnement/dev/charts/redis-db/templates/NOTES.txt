1. Get the Redis connection information:

{{- if contains "NodePort" .Values.service.type }}
  export NODE_PORT=$(kubectl get --namespace {{ .Release.Namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services {{ include "redis.fullname" . }})
  export NODE_IP=$(kubectl get nodes --namespace {{ .Release.Namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "Redis available at $NODE_IP:$NODE_PORT"
{{- else if contains "LoadBalancer" .Values.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status by running 'kubectl get --namespace {{ .Release.Namespace }} svc -w {{ include "redis.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc --namespace {{ .Release.Namespace }} {{ include "redis.fullname" . }} --template "{{"{{ range (index .status.loadBalancer.ingress 0) }}{{.}}{{ end }}"}}")
  echo "Redis available at $SERVICE_IP:{{ .Values.service.port }}"
{{- else if contains "ClusterIP" .Values.service.type }}
  echo "Redis is available within the cluster at:"
  echo "  {{ include "redis.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.service.port }}"
  echo ""
  echo "To connect from outside the cluster, run:"
  echo "  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "redis.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}"
{{- end }}

2. Connect to Redis:

{{- if .Values.redis.auth.enabled }}
  # With authentication
  kubectl run {{ include "redis.fullname" . }}-client --rm --tty -i --restart='Never' \
    --namespace {{ .Release.Namespace }} \
    --image redis:{{ .Values.image.tag | default .Chart.AppVersion }} \
    --command -- redis-cli -h {{ include "redis.fullname" . }} -a {{ .Values.redis.auth.password }}
{{- else }}
  # Without authentication
  kubectl run {{ include "redis.fullname" . }}-client --rm --tty -i --restart='Never' \
    --namespace {{ .Release.Namespace }} \
    --image redis:{{ .Values.image.tag | default .Chart.AppVersion }} \
    --command -- redis-cli -h {{ include "redis.fullname" . }}
{{- end }}

3. Test Redis connection:

  # Simple ping test
  kubectl exec -it deployment/{{ include "redis.fullname" . }} --namespace {{ .Release.Namespace }} -- redis-cli{{- if .Values.redis.auth.enabled }} -a {{ .Values.redis.auth.password }}{{- end }} ping

4. Redis details:
  - Service: {{ include "redis.fullname" . }}.{{ .Release.Namespace }}.svc.cluster.local:{{ .Values.service.port }}
{{- if .Values.redis.auth.enabled }}
  - Authentication: Enabled
{{- else }}
  - Authentication: Disabled
{{- end }}
  - Persistence: {{ if .Values.redis.persistence.enabled }}Enabled ({{ .Values.redis.persistence.size }}){{ else }}Disabled{{ end }}
